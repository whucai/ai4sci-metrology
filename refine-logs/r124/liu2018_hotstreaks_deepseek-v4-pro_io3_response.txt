# Preprocess: group by career id
def prepare_careers(df):
    careers = []
    for cid, group in df.groupby('id'):
        sorted_group = group.sort_values('work_index')
        impacts = sorted_group['impact'].values
        years = sorted_group['year'].values
        N = len(impacts)
        ranks = np.argsort(np.argsort(-impacts))  # rank 1 = highest impact
        top1_idx = np.argmax(impacts)
        # 2nd highest
        sorted_impacts = np.sort(impacts)[::-1]
        if len(sorted_impacts) >= 2:
            top2_val = sorted_impacts[1]
            top2_idx = np.where(impacts == top2_val)[0][0]
        else:
            top2_idx = top1_idx
        if len(sorted_impacts) >= 3:
            top3_val = sorted_impacts[2]
            top3_idx = np.where(impacts == top3_val)[0][0]
        else:
            top3_idx = top1_idx
        careers.append({
            'id': cid,
            'N': N,
            'impacts': impacts,
            'years': years,
            'top1': top1_idx,
            'top2': top2_idx,
            'top3': top3_idx,
            'ranks': ranks
        })
    return careers

def compute_random_impact_test(careers, label):
    """Test whether top hits are uniformly distributed (chi-square over 10 bins)."""
    bins = 10
    def test_for_rank(rank):
        positions = []
        for c in careers:
            idx = np.where(c['ranks'] == rank)[0]
            if len(idx) > 0:
                positions.append(idx[0] / c['N'])
        positions = np.array(positions)
        hist, edges = np.histogram(positions, bins=bins, range=(0,1))
        # expected uniform
        expected = np.ones(bins) * len(positions) / bins
        chi2, p = chi2_contingency([hist, expected])[:2] if len(positions)>bins else (0,1)
        return p
    for rank, name in [(1,'highest'), (2,'second'), (3,'third')]:
        p = test_for_rank(rank)
        print(f"RESULT random_impact_{name}_pvalue_{label} = {p:.4f}" +
              (f" SYNTHETIC" if SYNTHETIC else ""))

def phi_matrix(careers, label, bins=10):
    """Compute φ(N*,N**) normalized joint probability."""
    N1 = []
    N2 = []
    for c in careers:
        if c['N'] < 3:
            continue
        # positions of top1 and top2 normalized
        n1 = c['top1'] / c['N']
        n2 = c['top2'] / c['N']
        N1.append(n1)
        N2.append(n2)
    N1 = np.array(N1)
    N2 = np.array(N2)
    joint_hist, xedges, yedges = np.histogram2d(N1, N2, bins=bins, range=[[0,1],[0,1]])
    # marginal densities
    pN1 = np.histogram(N1, bins=bins, range=(0,1))[0] / len(N1)
    pN2 = np.histogram(N2, bins=bins, range=(0,1))[0] / len(N2)
    phi = np.zeros((bins, bins))
    for i in range(bins):
        for j in range(bins):
            if pN1[i]*pN2[j] > 0:
                phi[i,j] = joint_hist[i,j] / (len(N1) * pN1[i] * pN2[j])
    diag_mean = np.mean(np.diag(phi))
    print(f"RESULT phi_diagonal_mean_{label} = {diag_mean:.4f}" +
          (f" SYNTHETIC" if SYNTHETIC else ""))
    return phi

def R_measure(careers, label):
    """Compute R(ΔN/N) and its peak around zero."""
    diffs = []
    Ns = []
    for c in careers:
        diffs.append(abs(c['top1'] - c['top2']))  # absolute difference
        Ns.append(c['N'])
    diffs = np.array(diffs)
    Ns = np.array(Ns)
    frac = diffs / Ns
    # simple histogram of frac
    bins = np.linspace(0, 1, 21)
    P_emp, _ = np.histogram(frac, bins=bins, density=True)
    # shuffle within career to get null
    frac_shuf = []
    for c in careers:
        pos1 = np.random.choice(c['N'])
        pos2 = np.random.choice(c['N'])
        frac_shuf.append(abs(pos1-pos2)/c['N'])
    P_null, _ = np.histogram(frac_shuf, bins=bins, density=True)
    R = np.divide(P_emp, P_null, out=np.zeros_like(P_emp), where=P_null>0)
    # peak value around zero (first bin)
    peak = R[0] if R[0] > 0 else R[1]
    print(f"RESULT R_peak_{label} = {peak:.3f}" +
          (f" SYNTHETIC" if SYNTHETIC else ""))
    print(f"PAPER_REPORTED R_peak_{label} = {1.48 if label=='artists' else 1.42 if label=='directors' else 1.57}")
    return R

# Longest streak L
def longest_streak(careers, label):
    """Measure P(L) for real and shuffled careers."""
    L_real = []
    L_shuf = []
    for c in careers:
        impacts = c['impacts']
        threshold = np.median(impacts)
        above = impacts > threshold
        # real
        max_len = 0
        current_len = 0
        for a in above:
            if a:
                current_len += 1
                max_len = max(max_len, current_len)
            else:
                current_len = 0
        L_real.append(max_len)
        # shuffled
        shuf_above = np.random.permutation(above)
        max_len_s = 0
        current_len_s = 0
        for a in shuf_above:
            if a:
                current_len_s += 1
                max_len_s = max(max_len_s, current_len_s)
            else:
                current_len_s = 0
        L_shuf.append(max_len_s)
    # KS test between real and shuffled
    ks_stat, pval = ks_2samp(L_real, L_shuf)
    print(f"RESULT streak_KS_pvalue_{label} = {pval:.4f}" +
          (f" SYNTHETIC" if SYNTHETIC else ""))
    return L_real, L_shuf

# Hot‑streak estimation (simple maximum likelihood with two normal levels)
def fit_hot_streak(impacts, years):
    """Fit a single hot-streak model (Γ0, ΓH, t↑, t↓) to a career sequence.
       Returns parameters and log-likelihood."""
    N = len(impacts)
    best_ll = -np.inf
    best_params = None
    # try all possible start/end (coarse grid)
    for start in range(0, N-1):
        for end in range(start+1, min(N, start+20)):  # limit max length for speed
            # moving average for robust estimation? Use sample means
            pre = impacts[:start]
            in_streak = impacts[start:end+1]
            post = impacts[end+1:]
            if len(pre) == 0 and len(post) == 0:
                continue
            gamma0 = np.mean(np.concatenate([pre, post])) if len(pre)+len(post)>0 else 0
            gammaH = np.mean(in_streak)
            # Normal likelihood
            sigma = np.std(impacts) * 0.5  # fixed variance
            def nll(g0, gH, seg, sigma):
                return -np.sum(norm.logpdf(seg, loc=g0, scale=sigma))
            ll = -nll(gamma0, gamma0, pre, sigma) - nll(gammaH, gammaH, in_streak, sigma) - nll(gamma0, gamma0, post, sigma)
            if ll > best_ll:
                best_ll = ll
                best_params = (gamma0, gammaH, start, end)
    return best_params, best_ll
