def compute_rr(n_ret, n_all, scale=10000):
    """Retraction rate per `scale` articles."""
    return (n_ret / n_all) * scale

def wilson_ci(x, n, alpha=0.05):
    """Wilson score confidence interval for a proportion."""
    z = stats.norm.ppf(1 - alpha / 2)
    p_hat = x / n
    denom = 1 + z**2 / n
    center = (p_hat + z**2 / (2 * n)) / denom
    spread = z * np.sqrt((p_hat * (1 - p_hat) + z**2 / (4 * n)) / n) / denom
    return (center - spread, center + spread)

def katz_ci_mfrr(n_ret_m, n_all_m, n_ret_f, n_all_f, alpha=0.05):
    """Katz method (log-transform) CI for Male/Female Retraction Ratio."""
    p_m = n_ret_m / n_all_m
    p_f = n_ret_f / n_all_f
    # Variance of log(p) via delta method: (1-p)/(n*p)
    var_log_m = (1 - p_m) / (n_all_m * p_m)
    var_log_f = (1 - p_f) / (n_all_f * p_f)
    se_log = np.sqrt(var_log_m + var_log_f)
    z = stats.norm.ppf(1 - alpha / 2)
    log_mfrr = np.log((p_m / p_f))
    ci_log = (log_mfrr - z * se_log, log_mfrr + z * se_log)
    return tuple(np.exp(c) for c in ci_log)
