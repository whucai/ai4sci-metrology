import pandas as pd
import numpy as np

# ==============================================================================
# STUB DATA LOADING
# ==============================================================================
# The provided paper text is a stub and does not contain the actual dataset 
# or specific variable definitions. Based on the title "Technology transfer 
# measurement via patent/institutional linkage", we infer the required data 
# schema involves patents, institutions, and their linkages.
#
# REQUIRED DATA SCHEMA:
# 1. Patents Table:
#    - patent_id: Unique identifier for the patent.
#    - year: Grant or filing year.
#    - inventor_affiliations: List of institution IDs associated with inventors.
#    - assignee_affiliations: List of institution IDs associated with assignees.
#
# 2. Institutions Table:
#    - inst_id: Unique identifier for the institution.
#    - inst_type: Type of institution (e.g., 'University', 'Company', 'Government').
#
# SOURCE: Typically derived from patent databases (USPTO, EPO, PATSTAT) 
# combined with institutional registries.
# ==============================================================================

def load_stub_data():
    """
    Constructs a small synthetic/placeholder dataframe with the documented schema.
    """
    np.random.seed(42)
    
    # Synthetic Institutions
    institutions = pd.DataFrame({
        'inst_id': [1, 2, 3, 4, 5],
        'inst_name': ['Uni A', 'Uni B', 'Corp X', 'Corp Y', 'Gov Lab'],
        'inst_type': ['University', 'University', 'Company', 'Company', 'Government']
    })
    
    # Synthetic Patents with Linkages
    # We create patents that link institutions.
    # A "Technology Transfer" event is typically a patent linking a University and a Company.
    patents = pd.DataFrame({
        'patent_id': [101, 102, 103, 104, 105, 106, 107, 108],
        'year': [2020, 2020, 2021, 2021, 2022, 2022, 2023, 2023],
        'inventor_affiliations': [
            [1],           # Uni A only
            [1, 3],        # Uni A + Corp X (Linkage)
            [2],           # Uni B only
            [2, 4],        # Uni B + Corp Y (Linkage)
            [3],           # Corp X only
            [1, 2, 3],     # Uni A + Uni B + Corp X (Linkage)
            [5],           # Gov Lab only
            [1, 4]         # Uni A + Corp Y (Linkage)
        ],
        'assignee_affiliations': [
            [1],           # Assigned to Uni A
            [3],           # Assigned to Corp X (Transfer from Uni A)
            [2],           # Assigned to Uni B
            [4],           # Assigned to Corp Y (Transfer from Uni B)
            [3],           # Assigned to Corp X
            [3],           # Assigned to Corp X (Transfer from Unis)
            [5],           # Assigned to Gov Lab
            [4]            # Assigned to Corp Y (Transfer from Uni A)
        ]
    })
    
    return patents, institutions

# ==============================================================================
# ANALYSIS IMPLEMENTATION
# ==============================================================================
# The paper title suggests the core analysis is the "measurement" of technology 
# transfer via "patent/institutional linkage". 
# Since the full text is unavailable, we implement a standard metric for this 
# concept: Counting patents that exhibit a linkage between academic (University) 
# and industrial (Company) entities.
#
# INDICATOR: Technology Transfer Count (TTC)
# Formula: TTC = Sum over patents p of I(p has University Inventor AND Company Assignee)
# Note: This is a plausible interpretation of "patent/institutional linkage" 
# for technology transfer measurement.
# ==============================================================================

def analyze_technology_transfer(patents, institutions):
    """
    Performs the quantitative analysis based on the inferred metric.
    """
    # Merge institution types into patents for easier filtering
    # We need to check if a patent involves a University and a Company.
    
    # Helper to check if a list of IDs contains a specific type
    def has_type(affiliations, inst_df, target_type):
        if not affiliations:
            return False
        inst_types = inst_df.loc[inst_df['inst_id'].isin(affiliations), 'inst_type']
        return target_type in inst_types.values

    # Apply checks to each patent
    # Note: In a real scenario, this would be vectorized or done via SQL joins.
    # For the stub, we iterate for clarity.
    
    patents['has_uni_inventor'] = patents['inventor_affiliations'].apply(
        lambda x: has_type(x, institutions, 'University')
    )
    
    patents['has_corp_assignee'] = patents['assignee_affiliations'].apply(
        lambda x: has_type(x, institutions, 'Company')
    )
    
    # Define Technology Transfer Patent: 
    # A patent where a University is involved in invention and a Company is the assignee.
    # This represents the transfer of technology from academia to industry.
    patents['is_tech_transfer'] = patents['has_uni_inventor'] & patents['has_corp_assignee']
    
    # Calculate Metrics
    total_patents = len(patents)
    tech_transfer_patents = patents['is_tech_transfer'].sum()
    tech_transfer_rate = tech_transfer_patents / total_patents if total_patents > 0 else 0
    
    # Analysis by Year
    yearly_analysis = patents.groupby('year').agg(
        total=('patent_id', 'count'),
        tech_transfer=('is_tech_transfer', 'sum')
    )
    yearly_analysis['rate'] = yearly_analysis['tech_transfer'] / yearly_analysis['total']
    
    return {
        'total_patents': total_patents,
        'tech_transfer_patents': tech_transfer_patents,
        'tech_transfer_rate': tech_transfer_rate,
        'yearly_analysis': yearly_analysis
    }

# ==============================================================================
# MAIN EXECUTION
# ==============================================================================

if __name__ == "__main__":
    # 1. Load Data
    patents_df, institutions_df = load_stub_data()
    
    # 2. Run Analysis
    results = analyze_technology_transfer(patents_df, institutions_df)
    
    # 3. Print Results
    print("="*50)
    print("QUANTITATIVE ANALYSIS RESULTS")
    print("="*50)
    
    print(f"RESULT total_patents = {results['total_patents']}")
    print(f"RESULT tech_transfer_patents = {results['tech_transfer_patents']}")
    print(f"RESULT tech_transfer_rate = {results['tech_transfer_rate']:.4f}")
    
    print("\nYearly Breakdown:")
    print(results['yearly_analysis'])
    
    # 4. Conclusion
    # Since the paper text is a stub, we derive the conclusion from the 
    # implemented metric's logic.
    print("\n" + "="*50)
    print("CONCLUSION")
    print("="*50)
    print("The analysis measures technology transfer by identifying patents")
    print("that link University inventors with Company assignees.")
    print("The calculated rate represents the proportion of patents")
    print("indicating such institutional linkage.")
    
    # Comparison with Paper (Stub)
    # The paper text does not provide specific numbers to compare against.
    print("\nNOTE: The provided paper text is a stub. No PAPER_REPORTED")
    print("values are available for comparison.")
