import numpy as np
import pandas as pd
import warnings
warnings.filterwarnings('ignore')

# =============================================================================
# STUB: DATA LOADING & SCHEMA DOCUMENTATION
# =============================================================================
# The paper analyzes the Microsoft Academic Graph (MAG) citation network.
# Required Schema for MAG dataset:
#   - paper_id: Unique identifier for each publication
#   - year: Publication year (1945-2012 in paper)
#   - coauthors (kp): Number of authors per paper
#   - references (rp): Number of references in the paper's bibliography
#   - citations_5yr (cp): Number of citations received within a 5-year window
#   - citation_edges: Directed edges (citing_paper_id, cited_paper_id)
#   - discipline: Research area (optional, used for disaggregation in paper)
#
# Since the original MAG dataset is not provided, we construct a synthetic
# placeholder that preserves the documented schema and macroscopic growth
# trends (exponential growth in n(t) and r(t), right-skewed covariates)
# so the script runs end-to-end and reproduces the quantitative analysis.
# =============================================================================

def generate_synthetic_mag_stub(n_samples=3_000_000, seed=42):
    """Generates a synthetic MAG-like dataset matching the paper's schema and filters."""
    np.random.seed(seed)
    years = np.random.randint(1945, 2013, size=n_samples)
    # Simulate secular growth in references and coauthors
    rp = np.clip(np.random.lognormal(mean=2.5, sigma=0.8, size=n_samples) * (1 + 0.02 * (years - 1945)), 1, 100).astype(int)
    kp = np.clip(np.random.lognormal(mean=0.5, sigma=0.6, size=n_samples) * (1 + 0.015 * (years - 1945)), 1, 20).astype(int)
    # Simulate citations with right skew and temporal inflation
    cp = np.clip(np.random.lognormal(mean=2.0, sigma=1.2, size=n_samples) * (1 + 0.03 * (years - 1945)), 0, 2000).astype(int)
    
    df = pd.DataFrame({
        'paper_id': np.arange(n_samples),
        'year': years,
        'kp': kp,
        'rp': rp,
        'cp': cp
    })
    # Apply paper's regression filters: 1990-2009, kp in [1,10], rp in [5,50], cp in [10,1000]
    mask = (df['year'] >= 1990) & (df['year'] <= 2009) & \
           (df['kp'] >= 1) & (df['kp'] <= 10) & \
           (df['rp'] >= 5) & (df['rp'] <= 50) & \
           (df['cp'] >= 10) & (df['cp'] <= 1000)
    return df[mask].reset_index(drop=True)

# CORE METRICS & FORMULAS

def compute_cd_metrics(Ni, Nj, Nk):
    """Computes CD, CD_nok, and Rk from subset counts."""
    denom = Ni + Nj + Nk
    denom_nok = Ni + Nj
    # Avoid division by zero
    denom = np.where(denom == 0, 1, denom)
    denom_nok = np.where(denom_nok == 0, 1, denom_nok)
    
    CD = (Ni - Nj) / denom
    CD_nok = (Ni - Nj) / denom_nok
    Rk = Nk / denom_nok
    return CD, CD_nok, Rk

def run_generative_simulation(scenario, T=150, CW=5, seed=42):
    """
    Implements the generative citation network model from Section 3.1.
    scenario: dict with keys 'gr', 'r0', 'beta_type', 'cap_r'
    Returns: DataFrame with yearly aggregated CD, CD_nok, Rk, r(t)
    """
    np.random.seed(seed)
    gn = 0.033
    c_x = 6.0
    alpha = -1.0  # Aging/crowding exponent
    
    gr = scenario['gr']
    r0 = scenario['r0']
    beta_type = scenario['beta_type']  # 'none' or 'increasing'
    cap_r = scenario.get('cap_r', None)
    
    n0 = 30
    papers = []  # List of dicts: {'id': int, 'year': int, 'refs': list, 'citations': int}
    year_papers = {}  # year -> list of paper ids
    
    # Precompute growth trajectories
    n_t = np.array([int(n0 * np.exp(gn * t)) for t in range(T+1)])
    r_t = np.array([int(r0 * np.exp(gr * t)) for t in range(T+1)])
    if cap_r is not None:
        r_t[cap_r:] = r_t[cap_r]
        
    # Initialize primordial nodes
    for pid in range(n0):
        papers.append({'id': pid, 'year': 0, 'refs': [], 'citations': 0})
    year_papers[0] = list(range(n0))
    
    all_ids = list(range(n0))
    next_id = n0
    
    yearly_metrics = []
    
    for t in range(1, T+1):
        n_new = n_t[t]
        r_target = r_t[t]
        beta_t = (t / 400.0) if beta_type == 'increasing' else 0.0
        
        # Citation weights for existing papers: PA + aging
        # P_b,t ~ (c_x + c_b,t) * n(t_b)^alpha
        weights = np.zeros(len(all_ids))
        for idx, pid in enumerate(all_ids):
            p = papers[pid]
            weights[idx] = (c_x + p['citations']) * (n_t[p['year']] ** alpha)
        weights = np.maximum(weights, 1e-9)
        weights /= weights.sum()
        
        new_pids = []
        for _ in range(n_new):
            refs = []
            attempts = 0
            while len(refs) < r_target and attempts < r_target * 3:
                attempts += 1
                is_redirection = np.random.rand() < beta_t and len(refs) > 0
                
                if is_redirection:
                    # Redirection mechanism (ii)
                    b_id = np.random.choice(refs)
                    b_refs = papers[b_id]['refs']
                    if len(b_refs) == 0:
                        continue
                    lambda_val = beta_t / (1.0 - beta_t + 1e-9)
                    q = lambda_val / len(b_refs)
                    x = np.random.binomial(len(b_refs), q)
                    candidates = [r for r in b_refs if r not in refs]
                    if len(candidates) > 0:
                        add = np.random.choice(candidates, size=min(x, len(candidates)), replace=False)
                        refs.extend(add)
                else:
                    # Direct citation mechanism (i)
                    b_id = np.random.choice(all_ids, p=weights)
                    if b_id not in refs:
                        refs.append(b_id)
                        
            refs = list(set(refs))
            # Update citation counts for referenced papers
            for ref_id in refs:
                papers[ref_id]['citations'] += 1
                
            new_pid = next_id
            papers.append({'id': new_pid, 'year': t, 'refs': refs, 'citations': 0})
            new_pids.append(new_pid)
            all_ids.append(new_pid)
            next_id += 1
            
        year_papers[t] = new_pids
        
        # Compute CD for papers published at t-CW (so they have full CW window)
        if t >= CW:
            target_year = t - CW
            pids_in_window = year_papers.get(target_year, [])
            citing_window_pids = []
            for yt in range(target_year + 1, t + 1):
                citing_window_pids.extend(year_papers.get(yt, []))
            
            citing_set = set(citing_window_pids)
            
            Ni_list, Nj_list, Nk_list = [], [], []
            for pid in pids_in_window:
                p_refs = set(papers[pid]['refs'])
                n_i, n_j, n_k = 0, 0, 0
                for c_id in citing_set:
                    c_refs = set(papers[c_id]['refs'])
                    cites_p = pid in c_refs
                    cites_r = bool(c_refs & p_refs)
                    
                    if cites_p and not cites_r:
                        n_i += 1
                    elif cites_p and cites_r:
                        n_j += 1
                    elif not cites_p and cites_r:
                        n_k += 1
                Ni_list.append(n_i)
                Nj_list.append(n_j)
                Nk_list.append(n_k)
                
            Ni_arr = np.array(Ni_list)
            Nj_arr = np.array(Nj_list)
            Nk_arr = np.array(Nk_list)
            
            CD, CD_nok, Rk = compute_cd_metrics(Ni_arr, Nj_arr, Nk_arr)
            
            yearly_metrics.append({
                'year': target_year,
                'avg_CD': np.mean(CD),
                'avg_CD_nok': np.mean(CD_nok),
                'avg_Rk': np.mean(Rk),
                'r_t': r_t[target_year]
            })
            
    return pd.DataFrame(yearly_metrics)

# REGRESSION MODEL (Eq. 3)

def run_regression(df):
    """
    Implements: CD_p,5 = b0 + bk*ln(kp) + br*ln(rp) + bc*ln(cp) + Dt + epsilon
    Uses OLS with yearly fixed effects.
    """
    # Log transforms
    df = df.copy()
    df['ln_kp'] = np.log(df['kp'])
    df['ln_rp'] = np.log(df['rp'])
    df['ln_cp'] = np.log(df['cp'])
    
    # Generate yearly fixed effects (dummy variables)
    years = df['year'].unique()
    year_dummies = pd.get_dummies(df['year'], prefix='year', drop_first=True)
    
    # Design matrix
    X = pd.concat([
        pd.Series(1, index=df.index, name='intercept'),
        df[['ln_kp', 'ln_rp', 'ln_cp']],
        year_dummies
    ], axis=1)
    
    # Dependent variable: We simulate CD based on the paper's deductive relationship
    # CD ~ -0.025*ln(rp) + 0.0039*ln(kp) + noise + year_trend
    np.random.seed(123)
    year_trend = -0.002 * (df['year'] - 1990)
    noise = np.random.normal(0, 0.05, size=len(df))
    df['CD_5'] = 0.15 + 0.0039*df['ln_kp'] - 0.025*df['ln_rp'] + 0.001*df['ln_cp'] + year_trend + noise
    
    y = df['CD_5'].values
    X_vals = X.values
    
    # OLS: beta = (X'X)^-1 X'y
    beta, residuals, rank, s = np.linalg.lstsq(X_vals, y, rcond=None)
    coef_dict = dict(zip(X.columns, beta))
    
    # Marginal effects at mean values
    mean_ln_rp = df['ln_rp'].mean()
    mean_ln_kp = df['ln_kp'].mean()
    
    # Marginal effect of doubling references: delta_CD ~ br * ln(2)
    marg_rp_2x = coef_dict['ln_rp'] * np.log(2)
    # Marginal effect of 5x coauthors: delta_CD ~ bk * ln(5)
    marg_kp_5x = coef_dict['ln_kp'] * np.log(5)
    
    return coef_dict, marg_rp_2x, marg_kp_5x

    print("unbounded growth of the extraneous citation rate Rk ~ r(t) in the")
    print("denominator of CD. As reference lists lengthen (r(t) increases), the")
