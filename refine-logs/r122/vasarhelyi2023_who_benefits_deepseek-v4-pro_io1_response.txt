def generate_synthetic_data(seed=42):
    np.random.seed(seed)
    
    # Number of articles shared online in each area (from CEM matching counts)
    n_cs = 940      # 94 high-vis, 85 matched
    n_eng = 4440    # 444 high-vis, 431 matched
    n_ss = 6280     # 628 high-vis, 612 matched
    
    # True gender composition (manual labeling) proportions (from Table 2)
    # CS: FF=0.116, FM=0.157, MF=0.128, MM=0.599 (approx)
    # Eng: FF=0.105, FM=0.137, MF=0.156, MM=0.602
    # SS: FF=0.241, FM=0.100, MF=0.116, MM=0.543
    true_comp = {
        'Computer Science': {'FF':0.116, 'FM':0.157, 'MF':0.128, 'MM':0.599},
        'Engineering':      {'FF':0.105, 'FM':0.137, 'MF':0.156, 'MM':0.602},
        'Social Sciences':  {'FF':0.241, 'FM':0.100, 'MF':0.116, 'MM':0.543}
    }
    # Team size distribution (fewer than 10)
    team_size_probs = np.array([0.05, 0.1, 0.15, 0.2, 0.2, 0.15, 0.1, 0.04, 0.01])  # 1..9

        'Computer Science': {'mu_h': 1.4, 'sigma_h': 0.4, 'mu_if': 0.5, 'sigma_if': 0.5},
        'Engineering':      {'mu_h': 1.2, 'sigma_h': 0.5, 'mu_if': 0.2, 'sigma_if': 0.6},
        'Social Sciences':  {'mu_h': 1.0, 'sigma_h': 0.6, 'mu_if': 0.0, 'sigma_if': 0.8}
    }
    
    dfs = []
    for area, n in zip(['Computer Science', 'Engineering', 'Social Sciences'],
                       [n_cs, n_eng, n_ss]):
        comp = true_comp[area]
        # Draw true gender compositions
        types = list(comp.keys())
        probs = list(comp.values())
        true_types = np.random.choice(types, size=n, p=probs)
        first_gender_true = np.where(np.isin(true_types, ['FF','FM']), 'F', 'M')
        last_gender_true  = np.where(np.isin(true_types, ['FF','MF']), 'F', 'M')
        
        # Simulate algorithmic gender inference with errors
        # error rates roughly matching the paper (CS ~10% for women, Eng ~14%, SS ~3%)
        error_f_to_m = {'Computer Science':0.10, 'Engineering':0.14, 'Social Sciences':0.03}
        error_m_to_f = {'Computer Science':0.02, 'Engineering':0.02, 'Social Sciences':0.01}
        def flip(g, prob_f_to_m, prob_m_to_f):
            if g == 'F':
                return 'M' if np.random.rand() < prob_f_to_m else 'F'
            else:
                return 'F' if np.random.rand() < prob_m_to_f else 'M'
        first_gender_alg = np.array([flip(g, error_f_to_m[area], error_m_to_f[area]) for g in first_gender_true])
        last_gender_alg  = np.array([flip(g, error_f_to_m[area], error_m_to_f[area]) for g in last_gender_true])
        
        # Generate controls
        team_size = np.random.choice(np.arange(1,10), size=n, p=team_size_probs)
        max_h_index = np.exp(np.random.normal(area_params[area]['mu_h'], area_params[area]['sigma_h'], n))
        impact_factor = np.exp(np.random.normal(area_params[area]['mu_if'], area_params[area]['sigma_if'], n))
        # Add small positive shift to avoid zeros
        max_h_index = np.clip(max_h_index, 0.5, None)
        impact_factor = np.clip(impact_factor, 0.1, None)
        
        # Generate online shares (highly skewed)
        # log shares ~ a0 + a_h*h + a_if*if + a_size*size + noise
        log_shares_mean = (np.log(0.5) + 
                           0.3 * np.log(max_h_index) + 
                           0.2 * np.log(impact_factor) + 
                           0.1 * np.log(team_size) + 
                           np.random.normal(0, 0.8, n))
        shares = np.random.poisson(np.exp(log_shares_mean))
        shares = np.maximum(shares, 1)  # at least 1 share (Altmetric recorded)
        
        # Define high visibility as top 10% within area
        threshold = np.quantile(shares, 0.9)
        high_vis = (shares >= threshold).astype(int)
        
        # Generate citations such that the regression on matched data yields
        # coefficients similar to the paper.
        # We create log(cit+1) as a linear function of factors.
        # Coefficients (after CEM) are roughly:
        # CS: vis=0.404, FF=0.3, FM=0.25, MF=0.28, vis:FF=-0.3, vis:MF=-0.15
        # Eng: vis=0.216, FF=0.2, others ~0, vis:FF=0.1 (small)
        # SS: vis=1.309, FM=0.3, MF=0.3, others ~0, interactions ~0
        b0 = 0.5
        if area == 'Computer Science':
            b_vis = 0.404
            b_FF = 0.3; b_FM = 0.25; b_MF = 0.28
            b_int_FF = -0.3; b_int_FM = -0.05; b_int_MF = -0.15
        elif area == 'Engineering':
            b_vis = 0.216
            b_FF = 0.2; b_FM = 0.03; b_MF = 0.01
            b_int_FF = 0.1; b_int_FM = 0.0; b_int_MF = 0.0
        else:  # Social Sciences
            b_vis = 1.309
            b_FF = 0.1; b_FM = 0.3; b_MF = 0.3
            b_int_FF = 0.0; b_int_FM = 0.0; b_int_MF = 0.0
        
        # control coefficients (same across areas)
        c_h = 0.5; c_if = 0.3; c_size = 0.1
        # Use true gender dummies to generate citations (so that algorithmic errors add noise)
        FF_true = ((first_gender_true == 'F') & (last_gender_true == 'F')).astype(int)
        FM_true = ((first_gender_true == 'F') & (last_gender_true == 'M')).astype(int)
        MF_true = ((first_gender_true == 'M') & (last_gender_true == 'F')).astype(int)
        MM_true = 1 - FF_true - FM_true - MF_true
        
        log_cit_plus1 = (b0 + 
                         b_vis * high_vis +
                         b_FF * FF_true + b_FM * FM_true + b_MF * MF_true +
                         b_int_FF * high_vis * FF_true +
                         b_int_FM * high_vis * FM_true +
                         b_int_MF * high_vis * MF_true +
                         c_h * np.log(max_h_index) +
                         c_if * np.log(impact_factor) +
                         c_size * np.log(team_size) +
                         np.random.normal(0, 0.3, n))
        citations_5yr = np.maximum(0, np.round(np.exp(log_cit_plus1) - 1)).astype(int)
        
        df_area = pd.DataFrame({
            'article_id': [f"{area[:3]}_{i}" for i in range(n)],
            'area': area,
            'year': 2012,
            'first_gender_true': first_gender_true,
            'last_gender_true': last_gender_true,
            'first_gender_alg': first_gender_alg,
            'last_gender_alg': last_gender_alg,
            'team_size': team_size,
            'max_h_index': max_h_index,
            'journal_impact_factor': impact_factor,
            'altmetric_shares': shares,
            'citations_5yr': citations_5yr,
            'high_visibility': high_vis
        })
        dfs.append(df_area)
    
    return pd.concat(dfs, ignore_index=True)

def team_composition(first_g, last_g):
    if first_g == 'F' and last_g == 'F': return 'FF'
    if first_g == 'F' and last_g == 'M': return 'FM'
    if first_g == 'M' and last_g == 'F': return 'MF'
    return 'MM'

def cem_match(df, treat_var, match_on, n_bins=4):
    """
    Coarsened Exact Matching: coarsen match_on columns into n_bins quantiles,
    then for each treated unit (treat_var==1), randomly select a control unit
    in the same strata without replacement.
    Returns matched DataFrame.
    """
    df = df.copy()
    for col in match_on:
        # coarsen into bins based on quantiles
        df[col + '_bin'] = pd.qcut(df[col], q=n_bins, duplicates='drop', labels=False)
    
    strata_cols = [c + '_bin' for c in match_on]
    df['stratum'] = df[strata_cols].apply(tuple, axis=1)
    
    treated = df[df[treat_var] == 1]
    control = df[df[treat_var] == 0]
    matched_pairs = []
    used_control_ids = set()
    
    for idx, row in treated.iterrows():
        stratum = row['stratum']
        pool = control[(control['stratum'] == stratum) & (~control.index.isin(used_control_ids))]
        if len(pool) > 0:
            match = pool.sample(1).iloc[0]
            used_control_ids.add(match.name)
            matched_pairs.append(row)
            matched_pairs.append(match)
    if len(matched_pairs) == 0:
        return pd.DataFrame()  # empty
    return pd.DataFrame(matched_pairs)

def run_regression(data, outcome, predictors):
    """Fit OLS and return coefficient table and p-values."""
    X = data[predictors]
    X = sm.add_constant(X)
    y = data[outcome]
    model = sm.OLS(y, X).fit()
    results = pd.DataFrame({
        'coef': model.params,
        'pval': model.pvalues,
        'sig': model.pvalues.apply(lambda p: '***' if p<0.001 else ('**' if p<0.01 else ('*' if p<0.05 else '')))
    })
    return results, model

# Here we simply generate new algorithmic labels by adding noise to true labels

for area in areas:

    valid_iters = 0
    
    for it in range(100):
        # Create a copy of true labels and then introduce random noise to simulate algorithmic error.
        # We'll use the same error rates as originally for the area.
        error_f_to_m = {'Computer Science':0.10, 'Engineering':0.14, 'Social Sciences':0.03}[area]

        temp_team = [team_composition(f,l) for f,l in zip(first_alg_temp, last_alg_temp)]
        
        # Now adjust the labels so that the final marginal distribution matches target_counts exactly.
        # We randomly select articles to reassign to meet the target proportions.
        current_counts = pd.Series(temp_team).value_counts()
        adjusted_team = temp_team.copy()
        # For each category, if deficit, randomly choose articles not currently in that category and change them.
        for cat in ['FF','FM','MF','MM']:
            deficit = target_counts.get(cat,0) - current_counts.get(cat,0)
            if deficit > 0:
                # Candidates: articles not in cat
                cand_idx = [i for i,t in enumerate(adjusted_team) if t != cat]
                # Randomly choose deficit number to change to cat
                change_idx = np.random.choice(cand_idx, size=int(deficit), replace=False)
                for idx in change_idx:
                    adjusted_team[idx] = cat
            elif deficit < 0:
                # surplus: change some of these to other categories that have deficit
                surp = -deficit
                cat_idx = [i for i,t in enumerate(adjusted_team) if t == cat]
                change_idx = np.random.choice(cat_idx, size=int(surp), replace=False)
                # reassign to a category with deficit (random among deficits)
                deficit_cats = [c for c,t in target_counts.items() if t > (pd.Series(adjusted_team).value_counts().get(c,0))]
                if not deficit_cats:
                    deficit_cats = [c for c in target_counts.keys() if c != cat]
                for idx in change_idx:
                    new_cat = np.random.choice(deficit_cats)
                    adjusted_team[idx] = new_cat
        # Create swapped dataset
        swapped = sub.copy()
        swapped['team_swapped'] = adjusted_team
        # Derive dummies
        for gcat in ['FF','FM','MF']:
            swapped[f'gend_{gcat}'] = (swapped['team_swapped'] == gcat).astype(int)

        # CEM match
        matched_sw = cem_match(swapped, treat_var='high_visibility', match_on=controls, n_bins=4)

        # Run Model 3
        pred3_sw = ['high_visibility'] + controls + ['gend_FF','gend_FM','gend_MF',

        X = sm.add_constant(matched_sw[pred3_sw])
        y = matched_sw['log_cit_plus1']
        try:
            mod = sm.OLS(y,X).fit()
        except:
            continue
        valid_iters += 1
        for var in pval_keys:
            if var in mod.params and mod.pvalues[var] < 0.05:
                sig_counts[var] += 1
    
    # Compute percentages
    pcts = {var: (sig_counts[var]/valid_iters*100) if valid_iters>0 else 0 for var in pval_keys}
    robustness_pcts[area] = pcts
    print(f"\n{area} (valid iterations: {valid_iters}):")
    for var in pval_keys:
        print(f"  {var}: significant in {pcts[var]:.0f}% of swapped datasets")
