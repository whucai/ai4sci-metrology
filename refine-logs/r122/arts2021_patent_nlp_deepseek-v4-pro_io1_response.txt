def load_award_data():
    """
    STUB: Loads award patents and builds matched control group.
    Required data (external):
    - USPTO/PATSTAT dataset containing for each granted US utility patent:
        * patent_id (str/int)
        * filing_date (datetime)
        * grant_date (datetime)
        * title, abstract, claims concatenated and cleaned into:
            * keywords: list of unique stemmed keywords (after tokenization, stopword removal,
              common-word removal, stemming). Example: ['polymerase','chain','reaction']
            * bigrams: list of unique bigrams (two consecutive cleaned words) from the same text.
            * trigrams: list of unique trigrams (three consecutive cleaned words).
        * primary_class (categorical, e.g., USPC class)
        * subclass_list: list of patent subclasses assigned to the patent
        * backward_citations: number of patents cited
        * cited_patents: list of cited patent IDs (or patent numbers)
        * cited_patent_classes: list of primary classes of cited patents (for originality)
        * forward_citations_10yr: number of citations received within 10 years of grant
        * citing_patent_classes: list of primary classes of citing patents (for generality)
        * triadic_family_id (optional, for rejection study)
    - Award linkage table:
        * patent_id matched to one of the seven awards (Nobel, Lasker, Turing, etc.)
        * award_year, award_type
    - For control group matching: all other patents not linked to awards,
      with the above schema.
    
    This stub constructs a tiny synthetic dataframe with realistic structure.
    It will randomly assign "award" status to a few patents and generate
    text-matched controls based on Jaccard similarity of keywords.
    """
    np.random.seed(42)
    n_patents = 100
    years = list(range(1980, 1985))
    # synthetic vocabulary of stemmed keywords
    vocab = [f'kw{i}' for i in range(200)]
    # generate patents
    data = {
        'patent_id': [f'P{1000+i}' for i in range(n_patents)],
        'filing_year': np.repeat(years, n_patents // len(years))[:n_patents],
        'primary_class': np.random.choice(['A','B','C','D'], n_patents),
        'keywords': [list(np.random.choice(vocab, size=np.random.randint(40, 80), replace=False)) for _ in range(n_patents)],
    }
    df = pd.DataFrame(data)
    # assign random filing date within the year
    df['filing_date'] = df['filing_year'].apply(lambda y: pd.Timestamp(f'{y}-01-01') + pd.Timedelta(days=np.random.randint(0,365)))
    df['grant_date'] = df['filing_date'] + pd.Timedelta(days=365*2)
    # generate bigrams and trigrams naively from ordered keywords? The paper uses the actual token order.
    # We'll simulate by taking consecutive pairs/triples from a random ordering.
    df['bigrams'] = df['keywords'].apply(lambda lst: [f'{lst[i]}-{lst[i+1]}' for i in range(len(lst)-1) if np.random.rand()<0.3])
    df['trigrams'] = df['keywords'].apply(lambda lst: [f'{lst[i]}-{lst[i+1]}-{lst[i+2]}' for i in range(len(lst)-2) if np.random.rand()<0.2])
    # keyword pairs (all combinations)
    df['word_pairs'] = df['keywords'].apply(lambda lst: [f'{a}_{b}' for a,b in combinations(sorted(lst),2)])
    # other columns
    df['n_subclasses'] = np.random.randint(1,8, n_patents)
    df['subclasses'] = df['n_subclasses'].apply(lambda n: [f'S{i}' for i in range(n)])
    df['backward_citations'] = np.random.randint(0, 50, n_patents)
    df['cited_patents'] = [list(np.random.choice(df['patent_id'].values, size=np.random.randint(0,20), replace=False)) for _ in range(n_patents)]
    df['cited_classes'] = df['cited_patents'].apply(lambda pats: [np.random.choice(['A','B','C','D']) for _ in pats])
    df['forward_citations_10yr'] = np.random.randint(0, 100, n_patents)
    df['citing_classes'] = [list(np.random.choice(['A','B','C','D'], size=np.random.randint(0,50), replace=True)) for _ in range(n_patents)]
    # award flag: 10% are award patents
    df['is_award'] = np.random.choice([0,1], n_patents, p=[0.9,0.1])
    # For control matching: we'll later match within same filing_year, using Jaccard of keywords
    return df

def jaccard_similarity(set1, set2):
    if not set1 or not set2:
        return 0.0
    return len(set1 & set2) / len(set1 | set2)

def match_controls_award(df):
    """
    For each award patent, select one control patent from non-award patents
    filed in the same year with highest Jaccard index based on keywords.
    Exclude same patent family (simplified: exclude same patent ID).
    Return a DataFrame with matched pairs (award and control rows).
    """
    award = df[df['is_award']==1].copy()
    pool = df[df['is_award']==0].copy()
    matched_controls = []
    for idx, row in award.iterrows():
        same_year = pool[pool['filing_year'] == row['filing_year']]
        if len(same_year) == 0:
            continue
        target_kw = set(row['keywords'])
        similarities = same_year['keywords'].apply(lambda x: jaccard_similarity(target_kw, set(x)))
        if similarities.max() == 0:
            continue
        best_idx = similarities.idxmax()
        control = same_year.loc[best_idx].copy()
        control['is_award'] = 0
        matched_controls.append(control)
    control_df = pd.DataFrame(matched_controls)
    return pd.concat([award, control_df], ignore_index=True)

def compute_text_metrics(df, all_patents_sorted):
    """
    Compute all text-based novelty and impact measures.
    df: the matched case-control dataframe (including all patents needed for baseline?).
    all_patents_sorted: full patent history sorted by filing_date, with columns:
        patent_id, filing_date, keywords, bigrams, trigrams, word_pairs.
    We need to compute first occurrence and reuse counts across the whole history.
    For simplicity, we'll use a small closed set but compute relative to all_patents_sorted.
    """
    # Precompute first occurrence dates for each ngram/word/combo
    first_appear_kw = {}
    first_appear_bigram = {}
    first_appear_trigram = {}
    first_appear_wp = {}

    # For reuse, count future patents reusing each ngram
    # We'll compute after building the timeline.
    # For each patent in df, we'll compute new_word = number of keywords not seen before its filing_date.
    # But the baseline includes all patents filed before 1980, we have none so ignore.
    # We'll simulate using the provided all_patents_sorted (which should include earlier patents if needed).
    # For synthetic data, we'll just compute based on the df itself as the whole universe.
    universe = all_patents_sorted.copy()
    universe = universe.sort_values('filing_date')

    # Build occurrence dictionaries
    for idx, row in universe.iterrows():
        fid = row['filing_date']
        for kw in row['keywords']:
            if kw not in first_appear_kw:
                first_appear_kw[kw] = fid
        for bg in row['bigrams']:
            if bg not in first_appear_bigram:
                first_appear_bigram[bg] = fid
        for tg in row['trigrams']:
            if tg not in first_appear_trigram:
                first_appear_trigram[tg] = fid
        for wp in row['word_pairs']:
            if wp not in first_appear_wp:
                first_appear_wp[wp] = fid

    # Reuse counts: number of future patents containing each ngram
    # We'll use a simple loop over universe. Since data is tiny, it's fine.
    reuse_kw = defaultdict(int)
    reuse_bigram = defaultdict(int)
    reuse_trigram = defaultdict(int)
    reuse_wp = defaultdict(int)
    for idx, row in universe.iterrows():
        for kw in row['keywords']:
            reuse_kw[kw] += 1
        for bg in row['bigrams']:
            reuse_bigram[bg] += 1
        for tg in row['trigrams']:
            reuse_trigram[tg] += 1
        for wp in row['word_pairs']:
            reuse_wp[wp] += 1

    # Now compute for each patent in df
    def calc_new_and_reuse(row):
        kw_set = set(row['keywords'])
        bg_set = set(row['bigrams'])
        tg_set = set(row['trigrams'])
        wp_set = set(row['word_pairs'])

        new_kw = [k for k in kw_set if first_appear_kw.get(k) == row['filing_date']]
        new_bg = [b for b in bg_set if first_appear_bigram.get(b) == row['filing_date']]
        new_tg = [t for t in tg_set if first_appear_trigram.get(t) == row['filing_date']]
        new_wp = [w for w in wp_set if first_appear_wp.get(w) == row['filing_date']]

        new_word = len(new_kw)
        new_bigram = len(new_bg)
        new_trigram = len(new_tg)
        new_word_comb = len(new_wp)

        new_word_reuse = sum(1 + reuse_kw[k] for k in new_kw)
        new_bigram_reuse = sum(1 + reuse_bigram[b] for b in new_bg)
        new_trigram_reuse = sum(1 + reuse_trigram[t] for t in new_tg)
        new_word_comb_reuse = sum(1 + reuse_wp[w] for w in new_wp)

        return pd.Series([new_word, new_bigram, new_trigram, new_word_comb,
                          new_word_reuse, new_bigram_reuse, new_trigram_reuse, new_word_comb_reuse])

    df[['new_word', 'new_bigram', 'new_trigram', 'new_word_comb',
        'new_word_reuse', 'new_bigram_reuse', 'new_trigram_reuse', 'new_word_comb_reuse']] = df.apply(calc_new_and_reuse, axis=1)

    # Cosine similarity measures: need vector representations.
    # Build vocabulary from all universe.
    all_keywords = sorted(set().union(*universe['keywords']))
    vocab_idx = {k:i for i,k in enumerate(all_keywords)}
    # Create TF vectors (count)
    vectors = []
    for kw_list in universe['keywords']:
        vec = np.zeros(len(all_keywords))
        for k in kw_list:
            vec[vocab_idx[k]] += 1
        vectors.append(vec)
    vectors = np.array(vectors)
    # filing dates for universe
    dates = universe['filing_date'].values
    # For each patent in df, compute backward_cosine: average cosine similarity with all patents filed in 5 years before.
    # and forward_cosine: average with patents filed in 5 years after.
    # Note: 5 years before = window before filing; use filing_date.
    from sklearn.metrics.pairwise import cosine_similarity

    def cosine_avg(pid, window='backward'):
        row = df[df['patent_id']==pid].iloc[0]
        fdate = row['filing_date']
        # find universe indices
        uni_idx = universe[universe['patent_id']==pid].index[0]
        vec_query = vectors[uni_idx].reshape(1,-1)
        if window == 'backward':
            mask = (dates < fdate) & (dates >= fdate - pd.Timedelta(days=5*365))
        else:
            mask = (dates > fdate) & (dates <= fdate + pd.Timedelta(days=5*365))
        if mask.sum() == 0:
            return 0.0
        vec_window = vectors[mask]
        sim = cosine_similarity(vec_query, vec_window)
        return np.mean(sim)

    df['backward_cosine'] = df['patent_id'].apply(lambda pid: cosine_avg(pid, 'backward'))
    df['forward_cosine'] = df['patent_id'].apply(lambda pid: cosine_avg(pid, 'forward'))
    df['1_backward_cosine'] = 1 - df['backward_cosine']
    # Avoid division by zero
    df['forward_backward_cosine'] = np.where(df['backward_cosine']==0, 0, df['forward_cosine']/df['backward_cosine'])
    # Standardize
    df['1_backward_cosine'] = (df['1_backward_cosine'] - df['1_backward_cosine'].mean()) / df['1_backward_cosine'].std()
    df['forward_backward_cosine'] = (df['forward_backward_cosine'] - df['forward_backward_cosine'].mean()) / df['forward_backward_cosine'].std()
    return df

def compute_traditional_metrics(df, universe):
    """
    Compute traditional measures based on patent classification and citations.
    universe: full patent dataset sorted by filing_date.
    """
    # Subclass combinations new and reuse
    def get_subclass_pairs(subclasses):
        return [f'{a}_{b}' for a,b in combinations(sorted(subclasses),2)]
    universe['subclass_pairs'] = universe['subclasses'].apply(get_subclass_pairs)
    first_appear_sc = {}
    reuse_sc = defaultdict(int)
    for idx, row in universe.iterrows():
        fid = row['filing_date']
        for pair in row['subclass_pairs']:
            if pair not in first_appear_sc:
                first_appear_sc[pair] = fid
            reuse_sc[pair] += 1
    df['new_subclass_comb'] = df['subclasses'].apply(lambda scs: sum(1 for p in get_subclass_pairs(scs) if first_appear_sc.get(p) == df.loc[df.index[np.where(df['patent_id']==row['patent_id'])[0][0]],'filing_date']))
    # For reuse: sum(1+reuse_count)
    df['new_subclass_comb_reuse'] = df['subclasses'].apply(
        lambda scs: sum(1 + reuse_sc[p] for p in get_subclass_pairs(scs) if first_appear_sc.get(p) == df.loc[df.index[np.where(df['subclasses']==scs)[0][0]],'filing_date'] if not df.loc[df.index[np.where(df['subclasses']==scs)[0][0]],'filing_date'].empty else False)
    # This is messy; better do row-wise:
    new_sc_reuse = []
    for _, row in df.iterrows():
        sc_pairs = get_subclass_pairs(row['subclasses'])
        reuse_val = 0
        for p in sc_pairs:
            if first_appear_sc.get(p) == row['filing_date']:
                reuse_val += 1 + reuse_sc[p]
        new_sc_reuse.append(reuse_val)
    df['new_subclass_comb_reuse'] = new_sc_reuse

    # Citation combinations
    universe['cit_pairs'] = universe['cited_patents'].apply(lambda lst: [f'{a}_{b}' for a,b in combinations(sorted(lst),2)])
    first_appear_cc = {}
    reuse_cc = defaultdict(int)
    for idx, row in universe.iterrows():
        fid = row['filing_date']
        for pair in row['cit_pairs']:
            if pair not in first_appear_cc:
                first_appear_cc[pair] = fid
            reuse_cc[pair] += 1
    df['new_cit_comb'] = df['cited_patents'].apply(lambda lst: sum(1 for p in combinations(sorted(lst),2) if first_appear_cc.get(f'{p[0]}_{p[1]}') == df.loc[df.index[df['cited_patents'].apply(lambda x: x==lst)][0],'filing_date']))
    # For reuse:
    new_cc_reuse = []
    for _, row in df.iterrows():
        pairs = [f'{a}_{b}' for a,b in combinations(sorted(row['cited_patents']),2)]
        val = 0
        for p in pairs:
            if first_appear_cc.get(p) == row['filing_date']:
                val += 1 + reuse_cc[p]
        new_cc_reuse.append(val)
    df['new_cit_comb_reuse'] = new_cc_reuse

    # Originality: 1 - Herfindahl of cited patent classes
    def originality(row):
        classes = row['cited_classes']
        if len(classes)==0:
            return 0
        shares = pd.Series(classes).value_counts(normalize=True)
        return 1 - (shares**2).sum()
    df['originality'] = df.apply(originality, axis=1)

    # New_tech_origins: new combinations between focal patent classes and classes of cited patents
    # A combination is a pair (focal_class, cited_class) new to the database.
    # We'll treat all patents in same filing year as novel (as paper does).
    universe['focal_classes'] = universe['primary_class']  # simplified: primary class only
    universe['combos_tech'] = universe.apply(lambda r: [f'{r["primary_class"]}_{c}' for c in r['cited_classes']], axis=1)
    first_tech = {}
    for idx, row in universe.iterrows():
        fyear = row['filing_year']
        for combo in row['combos_tech']:
            if combo not in first_tech:
                first_tech[combo] = fyear
    df['new_tech_origins'] = df.apply(lambda r: sum(1 for c in [f'{r["primary_class"]}_{cc}' for cc in r['cited_classes']] if first_tech.get(c, 9999) == r['filing_year']), axis=1)

    # Forward_cit: log(1+forward_citations_10yr)
    df['forward_cit'] = np.log1p(df['forward_citations_10yr'])
    # Generality: 1 - Herfindahl of citing patent classes
    def generality(row):
        classes = row['citing_classes']
        if len(classes)==0:
            return 0
        shares = pd.Series(classes).value_counts(normalize=True)
        return 1 - (shares**2).sum()
    df['generality'] = df.apply(generality, axis=1)

    return df

def logit_evaluation(df, depvar, measures, control_vars):
    """
    Run separate logistic regression for each measure (plus controls) and report
    coefficient, AUC, precision, recall, average marginal effect.
    Measures: list of column names.
    control_vars: list of control columns.
    """
    X_controls = pd.get_dummies(df[control_vars].astype(str), drop_first=True)
    y = df[depvar]
    results = {}
    for m in measures:
        X = pd.concat([df[[m]], X_controls], axis=1).dropna(axis=1) # ensure no missing
        X = X.astype(float)
        model = LogisticRegression(max_iter=1000).fit(X, y)
        y_pred = model.predict(X)
        y_prob = model.predict_proba(X)[:,1]
        auc = roc_auc_score(y, y_prob)
        prec = precision_score(y, y_pred)
        rec = recall_score(y, y_pred)
        # marginal effect at mean of X? Instead, we compute average marginal effect
        # dP/dx_i = beta_i * P*(1-P) approximated.
        # We'll compute average derivative using coefficient * avg(p*(1-p))
        p_hat = y_prob
        coef = model.coef_[0][0]  # coefficient for the measure
        avg_p1p = np.mean(p_hat * (1-p_hat))
        mfx = coef * avg_p1p * 100  # in percent
        results[m] = {
            'coef': coef,
            'AUC': auc,
            'precision': prec,
            'recall': rec,
            'marginal_effect_%': mfx
        }
        # print for this measure
        print(f"RESULT {depvar} ~ {m}: coef={coef:.4f}, AUC={auc:.3f}, precision={prec:.3f}, recall={rec:.3f}, marginal_effect_per_std%= {mfx:.2f}")
    return results
