def tokenize_and_clean(text):
    text = text.lower()
    tokens = re.findall(r'[a-z0-9][a-z0-9-]*[a-z0-9]+|[a-z0-9]', text)
    tokens = [t for t in tokens if not t.isdigit() and len(t) > 1 and t not in STOPWORDS]
    return tokens

def stem_tokens(tokens):
    # Simple rule-based stemmer to avoid nltk dependency issues in offline env
    suffixes = ["ing", "tion", "sion", "ment", "ness", "able", "ible", "ful", "less", "ous", "ive", "al", "ic", "er", "est", "ed", "s", "y"]
    stemmed = []
    for t in tokens:
        for suf in suffixes:
            if t.endswith(suf) and len(t) - len(suf) > 2:
                t = t[:-len(suf)]
                break
        stemmed.append(t)
    return stemmed

def compute_all_metrics(df):
    df = df.copy()
    # Process text
    df["keywords"] = df.apply(lambda r: stem_tokens(tokenize_and_clean(r["title"] + " " + r["abstract"] + " " + r["claims"])), axis=1)
    df["keyword_set"] = df["keywords"].apply(set)
    df["num_words"] = df["keywords"].apply(len)
    
    # Global vocabulary & first appearance tracking
    all_kws = {}
    all_bigrams = {}
    all_trigrams = {}
    all_combos = {}
    
    # Sort by filing date
    df = df.sort_values("filing_date").reset_index(drop=True)
    
    # Precompute TF vectors for cosine similarity
    vocab = sorted(set(kw for kws in df["keywords"] for kw in kws))
    vocab_idx = {w: i for i, w in enumerate(vocab)}
    tf_matrix = np.zeros((len(df), len(vocab)))
    for i, kws in enumerate(df["keywords"]):
        counts = Counter(kws)
        for kw, cnt in counts.items():
            if kw in vocab_idx:
                tf_matrix[i, vocab_idx[kw]] = cnt
    # L2 normalize for cosine
    norms = np.linalg.norm(tf_matrix, axis=1, keepdims=True)
    norms[norms == 0] = 1
    tf_norm = tf_matrix / norms
    
    # Initialize metrics
    metrics = pd.DataFrame(index=df.index)
    metrics["new_word"] = 0
    metrics["new_word_reuse"] = 0
    metrics["new_bigram"] = 0
    metrics["new_bigram_reuse"] = 0
    metrics["new_trigram"] = 0
    metrics["new_trigram_reuse"] = 0
    metrics["new_word_comb"] = 0
    metrics["new_word_comb_reuse"] = 0
    metrics["backward_cosine"] = 0.0
    metrics["forward_cosine"] = 0.0
    metrics["1-backward_cosine"] = 0.0
    metrics["forward_backward_cosine"] = 0.0
    
    # Traditional metrics placeholders
    metrics["new_subclass_comb"] = 0
    metrics["new_subclass_comb_reuse"] = 0
    metrics["new_cit_comb"] = 0
    metrics["new_cit_comb_reuse"] = 0
    metrics["originality"] = 0.0
    metrics["new_tech_origins"] = 0
    metrics["forward_cit"] = 0
    metrics["generality"] = 0.0
    
    # Track first appearances
    first_kw = {}
    first_big = {}
    first_tri = {}
    first_comb = {}
    first_sub_comb = {}
    first_cit_comb = {}
    
    # Forward reuse counters (will be filled in second pass or accumulated)
    kw_reuse = Counter()
    big_reuse = Counter()
    tri_reuse = Counter()
    comb_reuse = Counter()
    sub_comb_reuse = Counter()
    cit_comb_reuse = Counter()
    
    # First pass: identify new items & build TF/cosine
    for i, row in df.iterrows():
        year = row["filing_date"]
        kws = row["keywords"]
        kws_set = row["keyword_set"]
        
        # Unigrams
        new_uni = [kw for kw in kws if kw not in first_kw]
        for kw in new_uni:
            first_kw[kw] = year
        metrics.loc[i, "new_word"] = len(new_uni)
        
        # Bigrams
        bigs = [f"{kws[j]}_{kws[j+1]}" for j in range(len(kws)-1)]
        new_bigs = [b for b in bigs if b not in first_big]
        for b in new_bigs:
            first_big[b] = year
        metrics.loc[i, "new_bigram"] = len(new_bigs)
        
        # Trigrams
        tris = [f"{kws[j]}_{kws[j+1]}_{kws[j+2]}" for j in range(len(kws)-2)]
        new_tris = [t for t in tris if t not in first_tri]
        for t in new_tris:
            first_tri[t] = year
        metrics.loc[i, "new_trigram"] = len(new_tris)
        
        # Word combinations (order independent)
        combos = [tuple(sorted(c)) for c in itertools.combinations(kws_set, 2)]
        new_combos = [c for c in combos if c not in first_comb]
        for c in new_combos:
            first_comb[c] = year
        metrics.loc[i, "new_word_comb"] = len(new_combos)
        
        # Subclass combinations
        subs = row["subclasses"]
        sub_combos = [tuple(sorted(c)) for c in itertools.combinations(subs, 2)]
        new_sub_combos = [c for c in sub_combos if c not in first_sub_comb]
        for c in new_sub_combos:
            first_sub_comb[c] = year
        metrics.loc[i, "new_subclass_comb"] = len(new_sub_combos)
        
        # Citation combinations
        cites = row["backward_citations"]
        cit_combos = [tuple(sorted(c)) for c in itertools.combinations(cites, 2)]
        new_cit_combos = [c for c in cit_combos if c not in first_cit_comb]
        for c in new_cit_combos:
            first_cit_comb[c] = year
        metrics.loc[i, "new_cit_comb"] = len(new_cit_combos)
        
        # Originality (1 - HHI of cited patent primary classes)
        # Simplified: assume cited patents have random classes for stub
        if len(cites) > 0:
            cited_classes = [f"C{np.random.randint(1,10)}" for _ in cites]
            class_counts = Counter(cited_classes)
            shares = np.array(list(class_counts.values())) / len(cites)
            hhi = np.sum(shares**2)
            metrics.loc[i, "originality"] = 1 - hhi
        else:
            metrics.loc[i, "originality"] = 0.0
            
        # New tech origins
        focal_classes = set([row["primary_class"]] + subs)
        if len(cites) > 0:
            cited_classes = set([f"C{np.random.randint(1,10)}" for _ in cites])
            new_origins = len(focal_classes & cited_classes) # Simplified proxy
            metrics.loc[i, "new_tech_origins"] = new_origins
        else:
            metrics.loc[i, "new_tech_origins"] = 0
            
        # Forward citations (within 10 years)
        fwd_cites = row["forward_citations"]
        metrics.loc[i, "forward_cit"] = sum(1 for fc in fwd_cites if fc["filing_date"] <= year + 10)
        
        # Generality (1 - HHI of citing patent classes)
        if len(fwd_cites) > 0:
            citing_classes = [f"C{np.random.randint(1,10)}" for _ in fwd_cites]
            class_counts = Counter(citing_classes)
            shares = np.array(list(class_counts.values())) / len(fwd_cites)
            hhi = np.sum(shares**2)
            metrics.loc[i, "generality"] = 1 - hhi
        else:
            metrics.loc[i, "generality"] = 0.0
            
        # Cosine similarity
        prev_mask = (df["filing_date"] >= year - 5) & (df["filing_date"] < year)
        next_mask = (df["filing_date"] > year) & (df["filing_date"] <= year + 5)
        
        if prev_mask.any():
            sims_prev = cosine_similarity(tf_norm[i:i+1], tf_norm[prev_mask])[0]
            metrics.loc[i, "backward_cosine"] = np.mean(sims_prev)
        if next_mask.any():
            sims_next = cosine_similarity(tf_norm[i:i+1], tf_norm[next_mask])[0]
            metrics.loc[i, "forward_cosine"] = np.mean(sims_next)
            
        # Accumulate reuse counts for future patents
        for kw in kws:
            kw_reuse[kw] += 1
        for b in bigs:
            big_reuse[b] += 1
        for t in tris:
            tri_reuse[t] += 1
        for c in combos:
            comb_reuse[c] += 1
        for c in sub_combos:
            sub_comb_reuse[c] += 1
        for c in cit_combos:
            cit_comb_reuse[c] += 1

    # Second pass: compute reuse metrics (only for items first introduced by this patent)
    for i, row in df.iterrows():
        year = row["filing_date"]
        kws = row["keywords"]
        kws_set = row["keyword_set"]
        
        # Reuse calculations
        new_uni = [kw for kw in kws if first_kw.get(kw) == year]
        metrics.loc[i, "new_word_reuse"] = sum(1 + kw_reuse[kw] for kw in new_uni)
        
        bigs = [f"{kws[j]}_{kws[j+1]}" for j in range(len(kws)-1)]
        new_bigs = [b for b in bigs if first_big.get(b) == year]
        metrics.loc[i, "new_bigram_reuse"] = sum(1 + big_reuse[b] for b in new_bigs)
        
        tris = [f"{kws[j]}_{kws[j+1]}_{kws[j+2]}" for j in range(len(kws)-2)]
        new_tris = [t for t in tris if first_tri.get(t) == year]
        metrics.loc[i, "new_trigram_reuse"] = sum(1 + tri_reuse[t] for t in new_tris)
        
        combos = [tuple(sorted(c)) for c in itertools.combinations(kws_set, 2)]
        new_combos = [c for c in combos if first_comb.get(c) == year]
        metrics.loc[i, "new_word_comb_reuse"] = sum(1 + comb_reuse[c] for c in new_combos)
        
        subs = row["subclasses"]
        sub_combos = [tuple(sorted(c)) for c in itertools.combinations(subs, 2)]
        new_sub_combos = [c for c in sub_combos if first_sub_comb.get(c) == year]
        metrics.loc[i, "new_subclass_comb_reuse"] = sum(1 + sub_comb_reuse[c] for c in new_sub_combos)
        
        cites = row["backward_citations"]
        cit_combos = [tuple(sorted(c)) for c in itertools.combinations(cites, 2)]
        new_cit_combos = [c for c in cit_combos if first_cit_comb.get(c) == year]
        metrics.loc[i, "new_cit_comb_reuse"] = sum(1 + cit_comb_reuse[c] for c in new_cit_combos)
        
        # Cosine derivatives
        bw = metrics.loc[i, "backward_cosine"]
        fw = metrics.loc[i, "forward_cosine"]
        metrics.loc[i, "1-backward_cosine"] = 1 - bw
        metrics.loc[i, "forward_backward_cosine"] = fw / bw if bw > 0 else 0.0
        
    # Log transform count variables as per paper
    log_vars = ["new_word", "new_word_reuse", "new_bigram", "new_bigram_reuse", 
                "new_trigram", "new_trigram_reuse", "new_word_comb", "new_word_comb_reuse",
                "new_subclass_comb", "new_subclass_comb_reuse", "new_cit_comb", "new_cit_comb_reuse",
                "new_tech_origins", "forward_cit"]
    for v in log_vars:
        metrics[f"log_{v}"] = np.log1p(metrics[v])
        
    return df.merge(metrics, left_index=True, right_index=True)

# =============================================================================
# MATCHING PROCEDURE
# =============================================================================
def match_patents(df, target_col, match_col, year_col="filing_date"):
    """Match each target patent to a control/rejected patent with highest Jaccard index in same year."""
    targets = df[df[target_col] == 1].copy()
    controls = df[df[target_col] == 0].copy()
    matched_pairs = []
    
    for _, t in targets.iterrows():
        t_year = t[year_col]
        t_kws = t["keyword_set"]
        candidates = controls[controls[year_col] == t_year]
        if candidates.empty:
            continue
        # Compute Jaccard
        jaccards = candidates.apply(lambda c: len(t_kws & c["keyword_set"]) / len(t_kws | c["keyword_set"]) if len(t_kws | c["keyword_set"]) > 0 else 0, axis=1)
        best_idx = jaccards.idxmax()
        matched_pairs.append((t.name, best_idx))
        
    return pd.DataFrame(matched_pairs, columns=["target_id", "control_id"])

# =============================================================================
# STATISTICAL ANALYSIS
# =============================================================================
def run_ttest_cohens_d(group1, group2, var):
    m1, m2 = group1[var].mean(), group2[var].mean()
    s1, s2 = group1[var].std(), group2[var].std()
    n1, n2 = len(group1), len(group2)
    pooled_std = np.sqrt(((n1-1)*s1**2 + (n2-1)*s2**2) / (n1+n2-2))
    cohens_d = (m1 - m2) / pooled_std if pooled_std > 0 else 0
    t_stat, p_val = stats.ttest_ind(group1[var], group2[var])
    return m1, m2, cohens_d, t_stat, p_val

def run_logit_analysis(df, y_col, metric_col, controls_cols):
    X = df[controls_cols + [metric_col]].copy()
    X = sm.add_constant(X)
    y = df[y_col]
    
    # Handle categorical controls if present
    if "primary_class" in controls_cols:
        X = pd.get_dummies(X, columns=["primary_class"], drop_first=True)
    if "filing_date" in controls_cols:
        X = pd.get_dummies(X, columns=["filing_date"], drop_first=True)
        
    model = sm.Logit(y, X)
    result = model.fit(disp=0, maxiter=1000)
    
    y_pred = result.predict(X)
    y_pred_bin = (y_pred > 0.5).astype(int)
    
    precision = precision_score(y, y_pred_bin, zero_division=0)
    recall = recall_score(y, y_pred_bin, zero_division=0)
    auc = roc_auc_score(y, y_pred)
    
    # Marginal effect: coef * std(x) * mean(y_pred*(1-y_pred)) * 100
    coef = result.params[metric_col]
    std_x = df[metric_col].std()
    me = coef * std_x * np.mean(y_pred * (1 - y_pred)) * 100
    
    return result, precision, recall, auc, me
