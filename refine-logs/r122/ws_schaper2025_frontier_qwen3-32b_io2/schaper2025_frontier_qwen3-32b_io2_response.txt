import numpy as np
import os

# =============================================================================
# 1. LOAD RAW DATA
# =============================================================================
data_path = "/workspace/raw_data/sciscinet_sample.parquet"

        "scientist_id": range(1000),

    "total_citations": ["total_citations", "citations", "total_cites", "cite_count"],
    "h_index": ["h_index", "h", "hindex", "h_idx"]
}

    if target not in df.columns:
        found = next((c for c in candidates if c in df.columns), None)

            df[target] = df[found]
