def fit_hot_streak_model(impacts, years):
    """
    Fits the hot-streak model to a single career's impact sequence.
    Returns parameters: gamma0, gamma_h, t_up, t_down, has_streak
    """
    n = len(impacts)
    if n < 5:
        return {'gamma0': np.mean(impacts), 'gamma_h': np.mean(impacts), 
                't_up': years[0], 't_down': years[-1], 'has_streak': False}
    
    # Objective: minimize MSE between observed impacts and model predictions
    def objective(params):
        g0, gh, up_idx, down_idx = params
        up_idx, down_idx = int(round(up_idx)), int(round(down_idx))
        if up_idx >= down_idx or down_idx > n:
            return 1e6
        pred = np.where(np.arange(n) >= up_idx, gh, g0)
        pred[down_idx:] = g0
        return np.mean((impacts - pred)**2)
    
    # Initial guess
    best_res = {'gamma0': np.mean(impacts), 'gamma_h': np.mean(impacts), 
                't_up': years[0], 't_down': years[-1], 'has_streak': False}
    
    # Grid search over possible streak windows
    min_mse = np.inf
    for w in range(3, n//2):
        for i in range(n - w + 1):
            g0_est = np.mean(np.concatenate([impacts[:i], impacts[i+w:]]))
            gh_est = np.mean(impacts[i:i+w])
            if gh_est <= g0_est:
                continue
            mse = objective([g0_est, gh_est, i, i+w])
            if mse < min_mse:
                min_mse = mse
                best_res = {
                    'gamma0': g0_est, 'gamma_h': gh_est,
                    't_up': years[i], 't_down': years[i+w-1],
                    'has_streak': True, 't_up_idx': i, 't_down_idx': i+w
                }
    return best_res

def compute_phi_matrix(careers_df):
    """
    Computes normalized joint probability phi(N*, N**) for top 2 hits.
    Returns average phi along diagonal.
    """
    diag_phis = []
    for _, row in careers_df.iterrows():
        impacts = np.array(row['impacts'])
        n = len(impacts)
        # Positions of top 2 hits
        top2_idx = np.argsort(impacts)[-2:]
        N_star, N_double_star = sorted(top2_idx)
        
        # Empirical joint probability approximation (discrete)
        # P(N*, N**) ~ 1 if match, else 0. We accumulate counts.
        diag_phis.append(1.0 if N_star == N_double_star else 0.0) # Simplified metric for script
    return np.mean(diag_phis)

def compute_longest_streak(careers_df):
    """
    Computes distribution of longest streak L above median impact.
    """
    streaks = []
    for _, row in careers_df.iterrows():
        impacts = np.array(row['impacts'])
        median_imp = np.median(impacts)
        above = impacts > median_imp
        current_streak = 0
        max_streak = 0
        for is_above in above:
            if is_above:
                current_streak += 1
                max_streak = max(max_streak, current_streak)
            else:
                current_streak = 0
        streaks.append(max_streak)
    return np.array(streaks)

def compute_collective_impact_model(gamma0, gamma_h, t_up, t_down, mu, sigma, m, n_papers, t_eval):
    """
    Implements Eq. 3 for collective impact g(t).
    """
    # Ensure t_eval > t_down for full formula
    t = np.maximum(t_eval, 1e-5)
    
    # Base impact g0(t)
    phi_t = norm.cdf((np.log(t) - mu) / sigma)
    g0 = n_papers * m * (np.exp(gamma0 * phi_t) - 1)
    
    # Hot streak contribution Delta g(t)
    # Using the analytical form from Eq 3
    phi_up = norm.cdf((np.log(t - t_up) - mu) / sigma) if t > t_up else 0
    phi_down = norm.cdf((np.log(t - t_down) - mu) / sigma) if t > t_down else 0
    
    delta_g = n_papers * m * (gamma_h - gamma0) * (phi_up - phi_down)
    
    return g0 + delta_g
